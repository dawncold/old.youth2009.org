选择排序（Python）

26Jan 
#! /usr/bin/env python
# coding: utf-8

def select_sort(arraylist):
   for i in range(0, len(arraylist)):
       minElement = i
       for j in range(i, len(arraylist)):
           if arraylist[j] < arraylist[minElement]:
               minElement = j
       temp = arraylist[i]
       arraylist[i] = arraylist[minElement]
       arraylist[minElement] = temp

if __name__ == "__main__":
   arr = [2, 424, 4, 13, 5, 99, 100, 231, 0, 1]
   print arr
   select_sort(arr)
   print arr
