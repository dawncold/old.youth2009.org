def merge(arraylist, first, middle, last):

temp = []
i = first
j = middle + 1
while i &lt;= middle and j &lt;= last:
    if arraylist[i] &lt;= arraylist[j]:
        temp.append(arraylist[i])
        i += 1
    else:
        temp.append(arraylist[j])
        j += 1
while i &lt;= middle:
    temp.append(arraylist[i])
    i += 1
while j &lt;= last:
    temp.append(arraylist[j])
    j += 1
for i in range(0, last - first + 1):
    arraylist[first + i] = temp[i]
